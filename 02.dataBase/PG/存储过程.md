CREATE OR REPLACE FUNCTION public.insertidlabeltime(objectid bigint, objectlabel character varying, starttime bigint, daycount integer)
 RETURNS SETOF refcursor
 LANGUAGE plpgsql
AS $function$
    declare v_counter int default 0;
    declare begintime varchar(50) ;
    declare endtime varchar(50) ;
	BEGIN
		begintime := clock_timestamp();
		raise notice 'begin=%',begintime;
		LOOP 
			INSERT INTO id_label_time (objectid,objectlabel,logtime ,value) VALUES (objectid,objectlabel,starttime,random()*(25-10)+10);
			v_counter := v_counter + 1;
			starttime := starttime + 60000;
			EXIT WHEN v_counter = daycount*24*60;
		END LOOP;
		endtime := clock_timestamp();
		raise notice 'end=%',endtime;
		raise notice '执行时长=%',to_timestamp(endtime,'yyyy/mm/dd hh24:mi:ss') - to_timestamp(begintime,'yyyy/mm/dd hh24:mi:ss');
	END;
$function$;
